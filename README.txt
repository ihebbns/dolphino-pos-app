  INSTALLATION RAPIDE (2 minutes)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

ÉTAPE 1 — Installer Node.js (une seule fois)
  → Télécharger sur : https://nodejs.org
  → Choisir la version "LTS" (ex: 20.x)
  → Installer avec les options par défaut

ÉTAPE 2 — Installer Dolphino POS
  → Double-cliquer sur : INSTALLER.bat
  → Attendre la fin de l'installation
  → L'application se lance automatiquement
  → Un raccourci est créé sur le Bureau

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  UTILISATION QUOTIDIENNE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  → Double-cliquer sur "Dolphino POS" sur le Bureau
    OU double-cliquer sur LANCER.bat

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  RACCOURCIS CLAVIER
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  F11           → Plein écran / Quitter plein écran
  Ctrl+P        → Imprimer le reçu
  Ctrl+R        → Recharger l'application
  Ctrl+= / -    → Zoom + / Zoom -
  Ctrl+0        → Réinitialiser le zoom
  F12           → Outils développeur
